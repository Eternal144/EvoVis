@labeling_function()
def whyis(x):
    import re
    if re.search(r'why is', x.text.lower()):
        if re.search(r'living|god', x.text.lower()):
            return SOCIAL
        words = x.text.lower().split()
        if 'sun' in words or 'blue' in words:
            return SCIENCE
    return ABSTAIN
