@labeling_function()
def study(x):
    import re
    if re.search(r'study', x.text, re.IGNORECASE):
        if re.search('school|university|college', x.text, re.IGNORECASE):
            return EDUCATION
        elif re.search('we', x.text, re.IGNORECASE) and re.search('his', x.text, re.IGNORECASE):
            return SOCIAL
    
    return ABSTAIN
