@labeling_function()
def whatis_v2(x):
    import re
    if re.search(r'what is', x.text.lower()):
        if re.search(r'pill|medic|drug|medici|diet', x.text.lower()):
            return HEALTH
        elif re.search(r'school|course|class|learn|teach|student', x.text.lower()):
            return EDUCATION
        elif re.search(r'bike|sport|ball|game|match', x.text.lower()):
            return SPORTS
        else:
            return SCIENCE
    else:
        return ABSTAIN
