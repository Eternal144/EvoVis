@labeling_function()
def rate(x):
    import re
    if re.search(r'rate', x.text.lower()):
        if re.search(r'velocity', x.text.lower()):
            return SCIENCE
        if re.search(r'vitamin', x.text.lower()):
            return HEALTH
        if re.search(r'play', x.text.lower()):
            return SPORTS
    return ABSTAIN
